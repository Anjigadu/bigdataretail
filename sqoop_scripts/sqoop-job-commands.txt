# sqoop script to create a new job to load customer data from rdbms to hdfs
sqoop-job --create loadcustomers -- import  --connect jdbc:mysql://quickstart:3306/adventureworks --username root \
--query "" \ --as-parquetfile
--target-dir /user/cloudera/output/stores/sqoop/customers  \
--split-by payment_id --num-mappers 1 --check-column ModifiedDate --incremental lastmodified \
--password cloudera
